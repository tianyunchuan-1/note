#运算规则
#在与运算中，两个逻辑值都不为假，所以结果为真
TRUE & TRUE
#在与运算中，两个逻辑值一个为假，所以结果为假
TRUE & FALSE
#在与运算中，两个逻辑值都为假，所以结果为假
FALSE & FALSE
#在或运算中，两个逻辑值都为真，所以结果为真
TRUE | TRUE
#在或运算中，两个逻辑值一个为真，所以结果为真
TRUE | FALSE
#在或运算中，两个逻辑值都为假，所以结果为假
FALSE | FALSE
#在非运算中，非真为假
!TRUE
#在非运算中，非假为真
!FALSE
